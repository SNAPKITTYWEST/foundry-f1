-- Lean 4.19.0 / Mathlib 4.19.0
-- Status: CLOSED — zero sorry
-- Tactic: constructor / rintro

theorem boole_idempotence
    (α : Type*) [BooleanAlgebra α] (x : α) :
    x ⊓ x = x ∧ x ⊔ x = x := by
  constructor
  · simp [inf_idem]
  · simp [sup_idem]
